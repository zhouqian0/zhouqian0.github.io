angular form 中的 disable() 导致 form valid 结果为 false

form 的 disable（）

在angular 的 form 中，如果整个 form 的状态变为 disabled，会使 form.valid 返回的值会变为 false，表单校验不通过（哪怕 form 的所有组件都是验证通过的）。

import {Component, OnInit} from '@angular/core';
import {FormGroup, FormControl, Validators} from '@angular/forms';
@Component({
  selector: 'my-app',
  templateUrl: './app.component.html',
  styleUrls: [ './app.component.css' ]
})
export class AppComponent implements OnInit{
  name = 'Angular';

  form = new FormGroup({
    'test':new FormControl('',Validators.required)
  });

  ngOnInit(){
    this.form.get('test').setValue('2');
    // this.form.disable();
    console.log(this.form.valid);
  }
}
